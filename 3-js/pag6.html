Aqui está um guia detalhado sobre cada um desses tópicos no JavaScript:

1. Funções de Captura de Dados (input, select, checkbox)

Para capturar valores do HTML, utilizamos document.querySelector ou document.getElementById.

Exemplo (Captura de input)

function capturarInput() {
  let nome = document.getElementById("nome").value;
  console.log("Nome digitado:", nome);
}
<input type="text" id="nome">
<button onclick="capturarInput()">Capturar</button>

Exemplo (Captura de select)

function capturarSelect() {
  let opcao = document.getElementById("opcoes").value;
  console.log("Opção selecionada:", opcao);
}
<select id="opcoes" onchange="capturarSelect()">
  <option value="A">Opção A</option>
  <option value="B">Opção B</option>
</select>

Exemplo (Captura de checkbox)

function capturarCheckbox() {
  let checkbox = document.getElementById("aceito").checked;
  console.log("Checkbox marcado?", checkbox);
}
<input type="checkbox" id="aceito" onchange="capturarCheckbox()"> Aceito os termos

2. Funções de Validação de Dados

Validação é essencial para garantir que os valores inseridos sejam corretos.

Exemplo (Validando Campo Obrigatório)

function validarNome() {
  let nome = document.getElementById("nome").value;
  if (nome.trim() === "") {
    alert("O campo Nome não pode estar vazio.");
    return false;
  }
  return true;
}

Exemplo (Validando Email)

function validarEmail(email) {
  let regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
}
console.log(validarEmail("teste@email.com")); // true
console.log(validarEmail("email errado")); // false

3. Funções de Conversão de Formatos

Essas funções são úteis para converter valores entre diferentes formatos.

Exemplo (Converter String para Número)

let numero = parseFloat("10.50");
console.log(numero); // 10.5

Exemplo (Converter Número para Moeda)

let valor = 1234.56;
console.log(valor.toLocaleString("pt-BR", { style: "currency", currency: "BRL" }));

Exemplo (Converter Data para Formato BR)

function formatarData(data) {
  let dataFormatada = new Date(data).toLocaleDateString("pt-BR");
  return dataFormatada;
}
console.log(formatarData("2024-02-12")); // "12/02/2024"

4. Funções de Modificação do DOM

Alteram elementos da página dinamicamente.

Exemplo (Alterar Texto de um Elemento)

function alterarTexto() {
  document.getElementById("titulo").innerText = "Texto alterado!";
}
<h1 id="titulo">Título Original</h1>
<button onclick="alterarTexto()">Alterar</button>

Exemplo (Alterar Cor de Fundo)

function alterarCor() {
  document.body.style.backgroundColor = "lightblue";
}

5. Funções de Reset

Redefinem valores de formulários.

Exemplo (Resetando um Formulário)

function resetarFormulario() {
  document.getElementById("meuFormulario").reset();
}
<form id="meuFormulario">
  <input type="text" name="nome">
  <button type="button" onclick="resetarFormulario()">Resetar</button>
</form>

6. Funções de Ocultação e Exibição de Informações no HTML

Escondem ou mostram elementos dinamicamente.

Exemplo (Mostrar e Ocultar Elemento)

function mostrarElemento() {
  document.getElementById("mensagem").style.display = "block";
}
        
function ocultarElemento() {
  document.getElementById("mensagem").style.display = "none";
}
<p id="mensagem" style="display: none;">Mensagem secreta!</p>
<button onclick="mostrarElemento()">Mostrar</button>
<button onclick="ocultarElemento()">Ocultar</button>

Exemplo (Alternar Visibilidade)

function alternarVisibilidade() {
  let elemento = document.getElementById("mensagem");
  elemento.style.display = (elemento.style.display === "none") ? "block" : "none";
}

7. Funções Chamando Funções

Uma função pode chamar outra.

Exemplo:

function mensagem() {
  return "Olá, ";
}
        
function saudacao(nome) {
  console.log(mensagem() + nome);
}
        
saudacao("Carlos"); // "Olá, Carlos"

8. Subfunções

Funções dentro de funções.

Exemplo:

function calcularPreco(preco) {
  function aplicarDesconto(valor) {
    return valor * 0.9;
  }
  return aplicarDesconto(preco);
}
        
console.log(calcularPreco(100)); // 90

9. Ordem de Escrita e Execução de Funções

O JavaScript executa funções de cima para baixo, e precisam ser chamadas.

Exemplo (Chamada Correta)

function mensagem() {
  console.log("Essa função foi chamada!");
}
mensagem(); // OK

Exemplo (Chamada Antes da Declaração - Não Funciona com const)

saudacao(); // ERRO se usar `const` ou `let`
        
function saudacao() {
  console.log("Oi!");
}

Funções anônimas precisam ser chamadas após serem declaradas:

const ola = function() {
  console.log("Oi!");
};
        
ola(); // OK
